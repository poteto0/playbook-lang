```playbook-lang
// 登場player
// pを抜いた文字が○の中に出てくる
// 1-5までしか許可されない
players = {
  p1, p2, p3, p4, p5
}

// 描画位置を設定する
state = {
  // ボーラーを設定
  baller = p1,

  // 初期位置を設定する
  position = {
    p1 = (x1, y1)
    p2 = (x2, y2)
    p3 = (x3, y3)
    p4 = (x4, y4)
    p5 = (x5, y5)
  },
}

// 動きを設定
action {
  // 動きを設定
  move = {
    p2 -> (xx2, yy2) // 座標を指定すると座標の位置に移動する
  },

  // スクリーンを設定
  screen = {
    p3 -> p2:before // 選手を設定すると、その選手が以前いた位置に移動する
  },

  // パス設定
  pass = {
    p1 -> p2:after // afterにすると移動後の位置にパスをする
  },
}
```
